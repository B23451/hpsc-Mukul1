# Use "git add <file>" or "git restore <file>" modified: filename
git diff <filename>
# In this situation above command will show the differences between
# both the files the second line of it tells git about the versions.
git diff HEAD <filename>
# This shows the difference between the most recently commited version
# of filename and current state of filename
git diff HEAD~1 <filename>
# This displayes the difference between current state of file and its 
# state 1 commit earlier
git show HEAD~2 <filename>
# This will show the 3rd last commit with commit-id, message along
# with it's difference with the current version
git diff f22b25e <filename>
# This is the commit id of any particular commit. This compares the state
# of file at that commit to the current state
git restore <filename>
# This option restores the current version of the file with most recently
# committed version
git restore -s f22b25e <filename> # We can get any version of file...
